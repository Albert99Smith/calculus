\author{Jim Fowler}

\outcome{Integrate a function defined on a rectangle}

\begin{document}

\begin{exercise}
  Let $R$ be the rectangle $[0,1] \times [1,2]$.
  
  Let $F : [0,1] \times [1,2] \to \R$ be the function defined by the rule
  \[
    F(x,y) = x^2 + y.
  \]
  Evaluate $\iint_R F \d A$.
  
  \begin{prompt}
    \[
      \iint_R F \d A = \answer{11/6}.
    \]
  \end{prompt}

\end{exercise}

\end{document}
