\begin{exercise}
Let $r(u) = 7 \left(-5 e^u-\frac{3}{\sqrt[5]{u}}\right) \sin (u)$. Compute:
\[
r'(u)
\begin{prompt}=\answer{7 \left(\frac{3}{5 u \sqrt[5]{u}}-5 e^u\right) \sin (u)+7 \left(-5 e^u-\frac{3}{\sqrt[5]{u}}\right) \cos (u)}\end{prompt}
\]
\end{exercise}
