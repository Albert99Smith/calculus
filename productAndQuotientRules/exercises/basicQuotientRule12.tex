\begin{exercise}
Let $c(\psi) = \frac{ 3 \sqrt{\psi }-3 \sin (\psi )}{-e^{\psi }}$. Compute:
\[
c'(\psi)
\begin{prompt}=\answer{e^{-\psi } \left(3 \sqrt{\psi }-3 \sin (\psi )\right)-e^{-\psi } \left(\frac{3}{2 \sqrt{\psi }}-3 \cos (\psi )\right)}\end{prompt}
\]
\end{exercise}
