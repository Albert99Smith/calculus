\outcome{Evaluate limits of functions of several variables.}
\outcome{Determine continuity of functions of several variables.} 

\newcommand{\point}[1]{\left(#1\right)} %this allows \point{ to be changed to \vector{ with a quick find and replace
\newcommand{\pt}[1]{\mathbf{#1}} %this allows \pt{ to be changed to \vec{ with a quick find and replace

\begin{document}
\begin{exercise}
  Consider $F(x,y) = \frac{xy^2-y^4}{x^2-y^4}$.  Compute the limit
  \[
  \lim_{\point{x,y}\to \point{4,2}} F(x,y) = \answer{1/2}
  \]
  (write \verb|DNE| if the limit does not exist)
  
  \begin{exercise}
  Is $F(x,y)$ continuous at $(4,2)$?
  \begin{multipleChoice}
    \choice{Yes}
    \choice[correct]{No}
  \end{multipleChoice}
  
\begin{feedback}[correct]
Note that although the limit exists, the function is not defined at $(4,2)$.
\end{feedback}
  \end{exercise}
\end{exercise}
