\begin{exercise}
  Consider $F(x,y) = \frac{6xy}{5x^2+4y^2}$. While computing
  \[
  \lim_{\vector{x,y}\to \vector{0,0}} F(x,y)
  \]
  a student provides the following response:
  \begin{image}
    \includegraphics{studentWork.png}
  \end{image}
  \begin{multipleChoice}
    \choice{The student's work is correct.}
    \choice[correct]{The student's work is incorrect.}
  \end{multipleChoice}
  \begin{exercise}
    Evaluate the limit along the pth $y=mx$.
    \[
    \lim_{\vector{x,y} \to \vector{x,mx}} F(x,y) = \answer{4/7}
    \]
    \begin{exercise}
      So we conclude:
      \begin{multipleChoice}
        \choice{The limit exists and is equal to $4/7$.}
        \choice[correct]{The limit does not exist.}
      \end{multipleChoice}
    \end{exercise}
  \end{exercise}
\end{exercise}
