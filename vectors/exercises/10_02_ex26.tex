\begin{exercise}
Find the unit vector $\uvec{u}$ in the second quadrant of $\R^2$ that
makes an angle of $2\pi/3$ radians with the $x$-axis.
\begin{prompt}
\[
\uvec{u}=\vector{\answer{\cos(\frac{2\pi}{3})},\answer{\sin(\frac{2\pi}{3})}}
\]
\end{prompt}

\end{exercise}
