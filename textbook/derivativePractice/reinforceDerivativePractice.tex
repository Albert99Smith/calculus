\begin{shuffle}
\begin{exercise}
Let $f(x)=-e^{-4 x-1} \sin (x)$.
\[
\ddx f(x) = \answer{-e^{-4 x-1} (\cos (x)-4 \sin (x))}
\]
\end{exercise}

\begin{exercise}
Let $f(x)=\frac{3}{x^7}+\sqrt[3]{x}+e^{-3 x}+\frac{1}{\sqrt[5]{x}}$.
\[
\ddx f(x) = \answer{-\frac{21}{x^8}-3 e^{-3 x}+\frac{1}{3 \sqrt[3]{x}^2}-\frac{1}{5 x \sqrt[5]{x}}}
\]
\end{exercise}

\end{shuffle}
