\section{Lines and curves in space}

\qrcode[height=3in]{//Lines on a Hyperboloid.bvr  
//hyperboloid
hyp=parametricSurface
hyp.setMinMax=s,-5,5
hyp.setMinMax=t,0,6.6
hyp.setEquation=x,(sqrt(1+s^2)*cos(t))
hyp.setEquation=y,(sqrt(1+s^2)*sin(t))
hyp.setEquation=z,(s)
hyp.setSegmentCount=30
hyp.spawn
//line1
line1=parametricCurve
line1.setMinMax=x,-3,3
line1.setEquation=x,(1/2+x*(-sqrt(3)))
line1.setEquation=y,(sqrt(3)/2+x)
line1.setEquation=z,(2*x)
line1.setWidth=.4,.4
line1.setSegmentCount=20
line1.color=red
line1.spawn
//line2
line2=parametricCurve

\section{Second derivative test}

\qrcode[height=3in}{//Hyperbolic Paraboloid Example.bvr
h=parametricSurface
h.setMinMax=s,-3,3
h.setMinMax=t,-3,3
h.setEquation=x,(s)
h.setEquation=y,(t)
h.setEquation=z,(s^2/3-t^2/3)
h.setSegmentCount=20
h.spawn


p1=parametricCurve
p1.setMinMax=t,-3,3
p1.setEquation=x,(t)
p1.setEquation=y,(0)

p1.setWidth=0.3,0.3
p1.color=blue
p1.spawn

p2=parametricCurve
