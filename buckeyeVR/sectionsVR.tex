hyp.setEquation=z,(s)
hyp.setSegmentCount=30
hyp.spawn
//line1
line1=parametricCurve
line1.setMinMax=x,-3,3
line1.setEquation=x,(1/2+x*(-sqrt(3)))
line1.setEquation=y,(sqrt(3)/2+x)
line1.setEquation=z,(2*x)
line1.setWidth=.4,.4
line1.setSegmentCount=20
line1.color=red
line1.spawn
//line2
