\begin{exercise}
Let $g(w) = 2 \sqrt{e^{\frac{\sin{w}}{3}}+1}$. Compute:

\[
\dd{w}g(w)
\begin{prompt}=\frac{1}{\sqrt{\answer{e^{\frac{\sin{w}}{3}}+1}}}e^{\answer{\frac{\sin{w}}{3}}}\cdot \frac{\answer{\cos{w}}}{3}\end{prompt}
\]
\end{exercise}
