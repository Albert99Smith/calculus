\begin{exercise}
Let $B(u) = -\frac{1}{\sqrt[5]{2} \sqrt[5]{\csc (2 u)}}$. Compute:

\[
\dd{u}B(u)
\begin{prompt}=-\frac{2^{\answer{4/5}}}{5}\cot (2 u)(\csc(2u))^{\answer{-\frac{1}{5}}}\end{prompt}
\]
\end{exercise}
