\begin{exercise}
Let $P(s) =4^{-3}\Bigl(\frac{2}{s^7}+3s \Bigr)^{4}$. Compute:

\[
\dd{s}P(s)
\begin{prompt}=4^{\answer{-2}}\Bigl(\frac{2}{s^7}+3s \Bigr)^{\answer{3}}\Bigl(\answer{-\frac{14}{s^8} +3}\Bigr)\end{prompt}
\]
\end{exercise}
