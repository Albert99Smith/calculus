\begin{exercise}
Let $a(k) = -\sin \left(3 \sqrt[4]{2} \sqrt[4]{-k}\right)$. Compute:

\[
\dd{k}a(k)
\begin{prompt}=\answer{\frac{3 \cos \left(3 \sqrt[4]{2} \sqrt[4]{-k}\right)}{2\ 2^{3/4} \sqrt[4]{-k}^3}}\end{prompt}
\]
\end{exercise}
