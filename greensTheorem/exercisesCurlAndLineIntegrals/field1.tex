\begin{exercise}
  Below we have plotted a discrete ``sampling'' of a vector field:
  \begin{image}
    \includegraphics{field1.png}
  \end{image}
  Let $C$ be a circle of radius $3$ centered at the origin drawn in a
  counterclockwise fashion.  What conclusions are definitely true?
  \begin{selectAll}
    \choice{This is definitely a gradient field.}
    \choice[correct]{This is definitely not a gradient field.}
    \choice{This field definitely has positive curl.}
    \choice{This field definitely has negative curl.}
    \choice{$\oint_C \vec{F}\dotp\d\vec{p} < 0$}
    \choice{$\oint_C \vec{F}\dotp\d\vec{p} = 0$}
    \choice[correct]{$\oint_C \vec{F}\dotp\d\vec{p} > 0$}
  \end{selectAll}
  \begin{hint}
    Note that the radius of the circle is irreverent. 
  \end{hint}
\end{exercise}
