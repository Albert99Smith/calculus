\begin{enumerate}

\item$ \ddx(\sin x)^{\tan{x}} = (\sin{x})^{\tan{x}}\answer{\Bigl(1+\sec^2{(x)}\ln{(\sin{x})}\Bigr)}$\\
\end{enumerate}
