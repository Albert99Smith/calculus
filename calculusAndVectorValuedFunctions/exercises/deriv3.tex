\begin{exercise}
  Consider $\vec{h}(t) = \vector{t^2,\cos(2t),\sin(t^3)}$.
  \[
  \vec{h}'(t) = \vector{2t,-2\sin(2t),3t^2\cos(t^3)}
  \]
\end{exercise}
