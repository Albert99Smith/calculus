\newcommand{\oddAns}{
\begin{multipleChoice}
\choice[correct]{$f$ is odd.}
\choice{$f$  is even.}
\choice{$f$ is neither.}
\end{multipleChoice}
}

\newcommand{\evenAns}{
\begin{multipleChoice}
\choice{$f$ is odd.}
\choice[correct]{$f$  is even.}
\choice{$f$ is neither.}
\end{multipleChoice}
}

\newcommand{\neitherAns}{
\begin{multipleChoice}
\choice{$f$ is odd.}
\choice{$f$  is even.}
\choice[correct]{$f$ is neither.}
\end{multipleChoice}
}

\begin{exercise}

For each of the following functions find its domain and determine whether the function is odd, even, or neither.


(A)  $f(x) = \frac{x}{\sqrt{16-x^2}}$\\
The domain of $f$ (from left to right) is $(\answer{-\infty},\answer{-4})\cup (\answer{-4},\answer{4})\cup (\answer{4},\answer{\infty})$.
\oddAns

(B) $f(x) = \frac{x^4}{\sqrt{16-x^2}}$\\
The domain of $f$ (from left to right) is  $(\answer{-\infty},\answer{-4})\cup (\answer{-4},\answer{4})\cup (\answer{4},\answer{\infty})$.
\evenAns

(C) $f(x) = \pi$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\evenAns

(D) $f(x) = e^{1-x^2} + 7$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\evenAns

(E) $f(x) = e^{1-x^2} + 7x$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\neitherAns

(F) $f(x) = e^{x} + 7x$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\neitherAns

(G) $f(x) = e^7 + 7x$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\neitherAns

(H) $f(x) = x^5 + 7x$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\oddAns

(I) $f(x) = \sin{x} + 7x$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\oddAns

(J) $f(x) = \sin^3{x}$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\oddAns

(K) $f(x) = \sin^{36}{x}$\\
The domain of $f$ is $(\answer{-\infty},\answer{\infty})$.
\evenAns


\end{exercise}
