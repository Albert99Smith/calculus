\begin{exercise}
  Let $\vec{F}(x,y,z) = \vector{y e^z-x^3,\sin (z)-y^3,x^3-z^3}$ and
  \[
  R = \{(x,y,z):  x^2+y^2+z^2 \le 9\}.
  \]
  Compute the flux of $\vec{F}$ across the surface of $R$.
  \begin{prompt}
  \[
  \oiint_{\partial R} \vec{F}\dotp\uvec{n} \d S = \answer{-2916\pi/5}
  \]
  \end{prompt}
\end{exercise}
