\begin{exercise}
  Suppose $\vec{v}, \vec{w} \in \R^3$ and further suppose that $\vec{v} \neq \vec{0}$ and $\vec{w} \neq \vec{0}$.  Let $\theta$ be the angle between $\vec{v}$ and $\vec{w}$.

  Which statements are necessarily true?
  \begin{selectAll}
    \choice[correct]{$\vec{v} \cross \vec{w} = - \vec{w} \cross \vec{v}$.}
    \choice{$\vec{v} \dotp \vec{w} = - \vec{w} \dotp \vec{v}$.}
    \choice[correct]{$\vec{v} \dotp \vec{w} = |\vec{v}| \cdot |\vec{w}| \cdot \cos \theta$.}
    \choice{$|\vec{v} \dotp \vec{w}| = |\vec{v}| \cdot |\vec{w}| \cdot \cos \theta$.}
    \choice[correct]{$\left( \vec{v} \dotp \vec{w} \right)^2 = |\vec{v}|^2 \cdot |\vec{w}|^2 \cdot \cos^2 \theta$.}
  \end{selectAll}
\end{exercise}
