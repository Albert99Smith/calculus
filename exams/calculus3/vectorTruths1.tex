\begin{exercise}
  Suppose $\vec{v}, \vec{w} \in \R^3$ and further suppose that $\vec{v} \neq \vec{0}$ and $\vec{w} \neq \vec{0}$.  Let $\theta$ be the angle between $\vec{v}$ and $\vec{w}$.

  Which statements are necessarily true?
  \begin{selectAll}
    \choice{$|3 \vec{v} + \vec{w}| = 3 \vec{v} + \vec{w}$.}
    \choice[correct]{If $\vec{v} = \vec{w}$ then $|\vec{u}| = |\vec{w}|$.}
    \choice{If $|\vec{v}| = |\vec{w}|$ then $\vec{u} = \vec{w}$.}
    \choice{$\vec{v} \cross \vec{w} = |\vec{v}| \cdot |\vec{w}| \cdot \sin \theta$.}
    \choice[correct]{$|\vec{v} \cross \vec{w}| = |\vec{v}| \cdot |\vec{w}| \cdot \sin \theta$.}
  \end{selectAll}    
\end{exercise}
