  Here we see several shaded regions:

    \resizebox{.2\textwidth}{!}{\includegraphics{region5a.png}} 
    \resizebox{.2\textwidth}{!}{\includegraphics{region6a.png}}
  \end{center}
  Below we have several iterated integrals. Each is integrating over one
  of the regions above. Identify which region goes with each integral.
  \begin{multicols}{2}
  \begin{align*}
    \answer[format=string]{B}&: \int_{-1}^{1} \int_{-1}^{x}F(x,y)\d y \d x\\
    \answer[format=string]{C}&: \int_{-1}^{1} \int_{-x^2}^{x^2}F(x,y)\d y \d x\\
    \answer[format=string]{E}&: \int_{-1}^{1} \int_{\sqrt{|x|}}^{1}F(x,y)\d y \d x\\
    \answer[format=string]{D}&: \int_{-1}^{1} \int_{x^2}^{1}F(x,y)\d y \d x\\
    \answer[format=string]{F}&: \int_{-1}^{1} \int_{-1}^{x^2}F(x,y)\d y \d x\\
    \answer[format=string]{A}&: \int_{-1}^{1} \int_{-|x|}^{|x|}F(x,y)\d y \d x\\
  \end{align*}
  \end{multicols}
\end{problem}

\begin{problem}
  \begin{selectAll}
    \choice[correct]{The function $F(x,y) = x^6+y^4$ has exactly one minimum.}
    \choice{If $F^{(1,0)}(1,2)=F^{(0,1)}(1,2)=0$, $F^{(2,0)}(1,2) >
      0$, $F^{(0,2)}(1,2)>0$, and $F^{(1,1)}(1,2) <0$, then $(1,2)$ is
      a local minimum for $F$.}
    \choice[correct]{If $F^{(1,0)}(3,4)=0$, $F^{(0,1)}(3,4)=0$,
      $F^{(2,0)}(3,4) =0$, $F^{(0,2)}(3,4)>0$, and $F^{(1,1)}(3,4)
      <0$, then $(3,4)$ is a saddle-point for $F$.}    
    \choice[correct]{If $F(x,y)$ has a critical point at $(-3,1)$, then
      $G(x,y)=\left(F(x,y)\right)^6$ also has a critical point at
      $(-3,1)$.}
    \choice[correct]{If $(2,-3)$ is a critical point of $F$, $F^{(1,1)}(2,-3) =
      0$, $F^{(2,0)}(2,-3)>0$, and $F^{(0,2)}(2,-3)<0$, then at
      $(2,-3)$ $F$ locally looks like a hyperbolic paraboloid.}
    \choice{None of these.}
  \end{selectAll}    
\end{problem}
