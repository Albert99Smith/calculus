\begin{exercise}
  Which statements are necessarily true?
  \begin{selectAll}
    \choice[correct]{If $G(x,y) = \int_0^x\int_0^y F(s,t)\d s \d t$,
      then $G^{(1,1)}(x,y) = F(x,y)$.}

    \choice{The chain rules states: $\dd{t} F(\vec{p}(t)) = \grad
      F(\vec{p}(t)) \dotp \grad F(\vec{p}'(t))$}

    \choice{If $F$ is a continuous function then $\int_3^8\int_2^7
      F(x,y) \d x \d y = \int_2^7\int_3^8 \d x\d y$.}

    \choice[correct]{If $(3,2)$ is a maximum of $F(x,y)$ under the
      constraint $G(x,y)=0$, then $|\grad F(3,2) \dotp\grad G(3,2)| =
      |\grad F(3,2)|\cdot |\grad G(3,2)|$.}
    
    \choice{The surface area of $z = x^2+y^2$ over some region $R$ is
      given by $\iint_R (x^2+y^2) \d A$.}

    \choice{$\int_0^1\int_0^x F(x,y) \d y \d x= \int_0^1\int_y^1
      F(x,y) \d y \d x$.}

    \choice[correct]{The integral
      $\int_{-1}^1\int_{-\sqrt{1-x^2}}^{\sqrt{1-x^2}} \d y \d x$ is
      the area of the unit disk.}

    \choice{$\int_0^1\int_x^1 F(x,y) \d x\d y = \int_0^1\int_y^1 F(x,y) \d y \d x$}

    \choice{Let $R=\{(x,y): x^2+y^2 \le 1\}$. If $\iint_R F(x,y) \d A
      =0$, then $F(x,y)=0$ for all $(x,y)$ in $R$.}

    \choice[correct]{If $F(x,y)= 10$, $\iint_R F(x,y)\d A$ is $10$ times the
      area of $R$.}
  \end{selectAll}
  \end{exercise}
