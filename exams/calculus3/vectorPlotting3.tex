\begin{exercise}

  What picture best represents $\vector{-2,-1}$?
  \begin{multipleChoice}
    \choice{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (3,2) -- (-1,-1);
      \end{tikzpicture}
    }
    \choice{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (1,2) -- (0,0);
      \end{tikzpicture}
    }
    \choice[correct]{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (2,1) -- (0,0);
      \end{tikzpicture}
    }    
  \end{multipleChoice}
\end{exercise}
