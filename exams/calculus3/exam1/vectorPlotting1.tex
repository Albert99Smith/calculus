\begin{exercise}

  What picture best represents $\vector{1,2}$?
  \begin{multipleChoice}
    \choice[correct]{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (0,0) -- (1,2);
      \end{tikzpicture}
    }
    \choice{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (-1,-1) -- (2,3);
      \end{tikzpicture}
    }
    \choice{
      \begin{tikzpicture}
        \draw[step=1cm,gray,very thin] (-1,-1) grid (3,3);
        \draw[->,ultra thick] (0,0) -- (2,1);
      \end{tikzpicture}
    }
  \end{multipleChoice}
\end{exercise}
