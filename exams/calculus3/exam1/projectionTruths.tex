\begin{exercise}
  Suppose $\vec{v}, \vec{w} \in \R^3$ and further suppose that $\vec{v} \neq \vec{0}$ and $\vec{w} \neq \vec{0}$.

  Which statements are necessarily true?
  \begin{selectAll}
    \choice[correct]{If $\proj_{\vec{v}} \vec{w} = \vec{w}$, then $\vec{w}$ and $\vec{v}$ are parallel.}
    \choice{If $\proj_{\vec{v}} \vec{w} = \vec{v}$, then $\vec{w}$ and $\vec{v}$ are parallel.}    
    \choice[correct]{$\scal_{\vec{v}} \vec{w} \leq |\vec{w}|$.}
    \choice[correct]{$\scal_{\vec{w}} \vec{w} = |\vec{w}|$.}
    \choice{$\scal_{\vec{v}} \vec{v} = 1.$}
  \end{selectAll}
\end{exercise}
